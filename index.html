-- Gralia V14 — Sistema de notificações -- Roda DEPOIS de gralia-v14-all-fixes.sql CREATE TABLE IF NOT EXISTS public.notifications ( id uuid PRIMARY KEY DEFAULT gen_random_uuid(), user_id uuid NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE, kind text NOT NULL CHECK (kind IN ( 'new_follower', 'follow_request', 'feedback_reply', 'imports_summary', 'legal_update_new' )), title text NOT NULL, body text, link_view text, link_param text, data jsonb, read_at timestamptz, created_at timestamptz DEFAULT now() ); CREATE INDEX IF NOT EXISTS notifications_user_unread_idx ON public.notifications(user_id, created_at DESC) WHERE read_at IS NULL; CREATE INDEX IF NOT EXISTS notifications_user_all_idx ON public.notifications(user_id, created_at DESC); ALTER TABLE public.notifications ENABLE ROW LEVEL SECURITY; DROP POLICY IF EXISTS "user_read_own_notifications" ON public.notifications; CREATE POLICY "user_read_own_notifications" ON public.notifications FOR SELECT TO authenticated USING (user_id = auth.uid()); DROP POLICY IF EXISTS "user_update_own_notifications" ON public.notifications; CREATE POLICY "user_update_own_notifications" ON public.notifications FOR UPDATE TO authenticated USING (user_id = auth.uid()) WITH CHECK (user_id = auth.uid()); -- Apagar notificações antigas (>60 dias) CREATE OR REPLACE FUNCTION public.cleanup_old_notifications() RETURNS int LANGUAGE plpgsql SECURITY DEFINER SET search_path = public AS $$ DECLARE v_count int; BEGIN DELETE FROM public.notifications WHERE created_at < (now() - interval '60 days'); GET DIAGNOSTICS v_count = ROW_COUNT; RETURN v_count; END; $$; -- Marcar todas como lidas CREATE OR REPLACE FUNCTION public.mark_all_notifications_read() RETURNS int LANGUAGE plpgsql SECURITY DEFINER SET search_path = public AS $$ DECLARE v_count int; BEGIN UPDATE public.notifications SET read_at = now() WHERE user_id = auth.uid() AND read_at IS NULL; GET DIAGNOSTICS v_count = ROW_COUNT; RETURN v_count; END; $$; GRANT EXECUTE ON FUNCTION public.mark_all_notifications_read() TO authenticated; -- TRIGGER: novo seguidor → cria notificação (só pra accepted) CREATE OR REPLACE FUNCTION public.notify_new_follower() RETURNS trigger LANGUAGE plpgsql SECURITY DEFINER SET search_path = public AS $$ DECLARE v_follower_username text; v_follower_name text; BEGIN IF NEW.status IS DISTINCT FROM 'accepted' THEN RETURN NEW; END IF; SELECT username, COALESCE(display_name, username) INTO v_follower_username, v_follower_name FROM public.profiles WHERE user_id = NEW.follower_id; IF v_follower_username IS NULL THEN RETURN NEW; END IF; INSERT INTO public.notifications (user_id, kind, title, body, link_view, link_param, data) VALUES ( NEW.followed_id, 'new_follower', '👋 ' || v_follower_name || ' começou a te seguir', NULL, 'profile', v_follower_username, jsonb_build_object('follower_id', NEW.follower_id, 'follower_username', v_follower_username) ); RETURN NEW; END; $$; DROP TRIGGER IF EXISTS notify_new_follower_trigger ON public.follows; CREATE TRIGGER notify_new_follower_trigger AFTER INSERT ON public.follows FOR EACH ROW EXECUTE FUNCTION public.notify_new_follower(); -- TRIGGER: pedido pendente → notificação pra dono CREATE OR REPLACE FUNCTION public.notify_follow_request() RETURNS trigger LANGUAGE plpgsql SECURITY DEFINER SET search_path = public AS $$ DECLARE v_requester_username text; v_requester_name text; BEGIN IF NEW.status IS DISTINCT FROM 'pending' THEN RETURN NEW; END IF; SELECT username, COALESCE(display_name, username) INTO v_requester_username, v_requester_name FROM public.profiles WHERE user_id = NEW.follower_id; IF v_requester_username IS NULL THEN RETURN NEW; END IF; INSERT INTO public.notifications (user_id, kind, title, body, link_view, data) VALUES ( NEW.followed_id, 'follow_request', '📬 ' || v_requester_name || ' quer te seguir', 'Aprove ou rejeite em Conta → Perfil público.', 'settings', jsonb_build_object('requester_id', NEW.follower_id, 'requester_username', v_requester_username) ); RETURN NEW; END; $$; DROP TRIGGER IF EXISTS notify_follow_request_trigger ON public.follows; CREATE TRIGGER notify_follow_request_trigger AFTER INSERT ON public.follows FOR EACH ROW EXECUTE FUNCTION public.notify_follow_request(); -- TRIGGER: admin respondeu feedback → notifica usuário CREATE OR REPLACE FUNCTION public.notify_feedback_reply() RETURNS trigger LANGUAGE plpgsql SECURITY DEFINER SET search_path = public AS $$ BEGIN IF NEW.admin_reply IS NOT NULL AND (OLD.admin_reply IS NULL OR OLD.admin_reply = '') THEN INSERT INTO public.notifications (user_id, kind, title, body, link_view, data) VALUES ( NEW.user_id, 'feedback_reply', '💬 Resposta da equipe ao seu feedback', LEFT(NEW.admin_reply, 200), 'settings', jsonb_build_object('feedback_id', NEW.id) ); END IF; RETURN NEW; END; $$; DROP TRIGGER IF EXISTS notify_feedback_reply_trigger ON public.user_feedback; CREATE TRIGGER notify_feedback_reply_trigger AFTER UPDATE ON public.user_feedback FOR EACH ROW EXECUTE FUNCTION public.notify_feedback_reply(); -- TRIGGER: nova atualização jurídica criada → broadcast pra todos -- Usa is_active (nome real da coluna) CREATE OR REPLACE FUNCTION public.notify_legal_update() RETURNS trigger LANGUAGE plpgsql SECURITY DEFINER SET search_path = public AS $$ BEGIN IF NEW.is_active = true THEN INSERT INTO public.notifications (user_id, kind, title, body, link_view, data) SELECT u.id, 'legal_update_new', '⚖️ Nova atualização: ' || LEFT(NEW.title, 80), LEFT(COALESCE(NEW.summary, ''), 200), 'legalUpdates', jsonb_build_object('update_id', NEW.id) FROM auth.users u WHERE u.email IS NOT NULL LIMIT 10000; END IF; RETURN NEW; END; $$; DROP TRIGGER IF EXISTS notify_legal_update_trigger ON public.legal_updates; CREATE TRIGGER notify_legal_update_trigger AFTER INSERT ON public.legal_updates FOR EACH ROW EXECUTE FUNCTION public.notify_legal_update(); -- ============================================================ -- AGREGADO DE IMPORTAÇÕES (notificação resumida diária) -- ============================================================ CREATE OR REPLACE FUNCTION public.aggregate_imports_notifications() RETURNS int LANGUAGE plpgsql SECURITY DEFINER SET search_path = public AS $$ DECLARE v_owner_id uuid; v_count int; v_total int := 0; v_yesterday timestamptz := date_trunc('day', now() - interval '1 day'); v_today timestamptz := date_trunc('day', now()); BEGIN -- Pra cada owner que teve recursos importados ontem FOR v_owner_id, v_count IN SELECT sr.owner_id, COUNT(*) FROM public.resource_imports ri JOIN public.shared_resources sr ON sr.id = ri.resource_id WHERE ri.imported_at >= v_yesterday AND ri.imported_at < v_today GROUP BY sr.owner_id LOOP IF NOT EXISTS ( SELECT 1 FROM public.notifications WHERE user_id = v_owner_id AND kind = 'imports_summary' AND created_at >= v_yesterday ) THEN INSERT INTO public.notifications (user_id, kind, title, body, link_view) VALUES ( v_owner_id, 'imports_summary', '📥 ' || v_count || ' ' || (CASE WHEN v_count = 1 THEN 'pessoa importou' ELSE 'pessoas importaram' END) || ' seu material', 'Resumo do dia anterior. Veja detalhes na descoberta.', 'discover' ); v_total := v_total + 1; END IF; END LOOP; RETURN v_total; END; $$; SELECT 'V14 notifications criadas' AS status;
